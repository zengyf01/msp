```python
def execute_mpc(self, task_id: str, inputs: dict, params: dict) -> dict:
    """执行MPC (安全多方计算)"""
    mpc_type = params.get("mpc_type", "addition")

    if mpc_type == "addition":
        result = self.spu_device.add(inputs['data_a'], inputs['data_b'])
    elif mpc_type == "multiplication":
        result = self.spu_device.mul(inputs['data_a'], inputs['data_b'])
    elif mpc_type == "comparison":
        result = self.spu_device.lt(inputs['data_a'], inputs['data_b'])

    plain_result = sf.reveal(result)

    return {
        "status": "ok",
        "result": plain_result,
        "mpc_type": mpc_type,
    }

# mpc_runner.py - _do_run 重写
def _do_run(self, task_id: str, inputs: dict, params: dict) -> dict:
    return self.secretflow_adapter.execute_mpc(task_id, inputs, params)
```

```python
# secretflow_adapter.py - 新增方法
def execute_vertical_fl(
    self,
    task_id: str,
    inputs: Dict[str, Any],
    params: Dict[str, str]
) -> Dict[str, Any]:
    """执行纵向联邦学习"""
    model_type = params.get("model_type", "logistic_regression")

    if model_type in ("secureboost", "sgb"):
        return self.train_sgb(inputs, params)
    elif model_type == "vertical_lr":
        return self.train_vertical_lr(inputs, params)
    else:
        return self.train_default(inputs, params)

def train_sgb(self, inputs: dict, params: dict) -> dict:
    """训练 SecureBoost 模型"""
    from secretflow.ml.boost import SGB

    sgb = SGB(
        device=self.spursu,
        num_boosting_rounds=int(params.get("num_trees", 10)),
        max_depth=int(params.get("max_depth", 6)),
    )

    model = sgb.train(
        data=inputs.get("data_sources", {}),
        label=params.get("label_column"),
        feature_names=inputs.get("feature_columns", []),
    )

    return {
        "status": "ok",
        "model_type": "secureboost",
        "num_trees": params.get("num_trees", 10),
    }
```

```python
def execute_psi(self, task_id: str, inputs: dict, params: dict) -> dict:
    """执行PSI (隐私集合求交)"""
    psi_type = params.get("psi_type", "ecdh")  # ecdh, kkrt, bc22

    if psi_type == "ecdh":
        from secretflow.preprocessing import PSI
        psi = PSI(self.spursu, key_column=params.get("key_column"))
        result = psi.run(data=inputs.get("data_source"))

    elif psi_type == "kkrt":
        from secretflow.preprocessing.psi import KKRTPSI
        psi = KKRTPSI(self.spursu, key_column=params.get("key_column"))
        result = psi.run(data=inputs.get("data_source"))

    elif psi_type == "bc22":
        from secretflow.preprocessing.psi import BC22PSI
        psi = BC22PSI(self.spursu, key_column=params.get("key_column"))
        result = psi.run(data=inputs.get("data_source"),
                         num_buckets=int(params.get("bucket_size", 1000)))

    return {
        "status": "ok",
        "psi_type": psi_type,
        "matched_count": len(result),
    }
```

```python
# dag_executor.py
class DAGExecutor:
    """DAG 执行器"""

    def __init__(self, dag_def: dict, adapter: SecretFlowAdapter):
        self.dag = dag_def
        self.adapter = adapter

    def execute(self) -> dict:
        # 1. 解析 DAG，生成执行计划
        execution_plan = self._plan_execution()

        # 2. 按拓扑顺序执行节点
        results = {}
        for node in execution_plan:
            result = self._execute_node(node, results)
            results[node['nodeId']] = result

        return {"status": "ok", "results": results}

    def _execute_node(self, node: DAGNode, upstream_results: dict) -> dict:
        # 调用 SecretFlow 的 comp_eval
        from secretflow.component.core import comp_eval

        # 构建输入
        inputs = self._build_inputs(node, upstream_results)

        # 调用组件
        res = comp_eval(
            param=node['comp_id'],
            storage_config=node['storage_config'],
            cluster_config=node['cluster_config'],
        )

        return res
```

```python
class CompoundRunner(BaseRunner):
    """复合任务运行器 - 协调 PSI+FL+MPC 三阶段"""

    def _do_run(self, task_id: str, inputs: dict, params: dict) -> dict:
        stages = params.get("stages", [])

        results = {}
        for stage in stages:
            if stage["type"] == "PSI":
                result = self.psi_runner.run(task_id, inputs, stage["params"])
            elif stage["type"] == "FL":
                result = self.fl_runner.run(task_id, inputs, stage["params"])
            elif stage["type"] == "MPC":
                result = self.mpc_runner.run(task_id, inputs, stage["params"])

            results[stage["order"]] = result

        return {"status": "completed", "stages": results}
```
